ef next_step(MAZE,ex,ey):

    elif MAZE[a][b-1]==0:
        push(a,b-1)
        MAZE[a][b-1]=2
    elif MAZE[a+1][b]==0:
        push(a+1,b)
        MAZE[a+1][b]=2
    elif MAZE[a][b+1]==0:
